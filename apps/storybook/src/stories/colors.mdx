import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs/blocks'

<Meta title="Styling & Theming/Colors" />

# Color Palette

## Primary Colors

<ColorPalette>
  {(() => {
    const styles = getComputedStyle(document.documentElement);
    const themeColors = {
      'primary': styles.getPropertyValue('--color-primary').trim(),
      'primary-foreground': styles.getPropertyValue('--color-primary-foreground').trim(),
      'secondary': styles.getPropertyValue('--color-secondary').trim(),
      'secondary-foreground': styles.getPropertyValue('--color-secondary-foreground').trim(),
      'accent': styles.getPropertyValue('--color-accent').trim(),
      'accent-foreground': styles.getPropertyValue('--color-accent-foreground').trim(),
    };

    return Object.entries(themeColors).map(([name, value]) => (
      <ColorItem
        key={name}
        title={name}
        subtitle={value}
        colors={{ [name]: value }}
      />
    ))

})()}

</ColorPalette>
